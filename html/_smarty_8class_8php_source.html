01533     function _fetch_resource_info(&$params)
01534     {
01535         if(!isset($params['get_source'])) { $params['get_source'] = true; }
01536         if(!isset($params['quiet'])) { $params['quiet'] = false; }
01537 
01538         $_return = false;

01568                         call_user_func_array($this->_plugins['resource'][$_resource_type][0][1],
01569                                              array($_resource_name, &$params['resource_timestamp'], &$this));
01570

01617     function _parse_resource_name(&$params)
01618     {
01619 
01620         // split tpl_path by the first colon
01621         $_resource_name_parts = explode(':', $params['resource_name'], 2);
01622 
01623         if (count($_resource_name_parts) == 1) {
01624             // no resource type given
01625             $params['resource_type'] = $this->default_resource_type;
01626             $params['resource_name'] = $_resource_name_parts[0];
01627         } else {
01628             if(strlen($_resource_name_parts[0]) == 1) {
01629                 // 1 char is not resource type, but part of filepath
01630                 $params['resource_type'] = $this->default_resource_type;
01631                 $params['resource_name'] = $params['resource_name'];
01632             } else {
01633                 $params['resource_type'] = $_resource_name_parts[0];
01634                 $params['resource_name'] = $_resource_name_parts[1];
01635             }
01636         }
01637 
01638         if ($params['resource_type'] == 'file') {
01639             if (!preg_match('/^([\/\\\\]|[a-zA-Z]:[\/\\\\])/', $params['resource_name'])) {
01640                 // relative pathname to $params['resource_base_path']
01641                 // use the first directory where the file is found
01642                 foreach ((array)$params['resource_base_path'] as $_curr_path) {
01643                     $_fullpath = $_curr_path . DIRECTORY_SEPARATOR . $params['resource_name'];
01644                     if (file_exists($_fullpath) && is_file($_fullpath)) {
01645                         $params['resource_name'] = $_fullpath;
01646                         return true;
01647                     }
01648                     // didn't find the file, try include_path
01649                     $_params = array('file_path' => $_fullpath);
01650                     require_once(SMARTY_CORE_DIR . 'core.get_include_path.php');
01651                     if(smarty_core_get_include_path($_params, $this)) {
01652                         $params['resource_name'] = $_params['new_file_path'];
01653                         return true;
01654                     }
01655                 }
01656                 return false;
01657             } else {
01658                 /* absolute path */
01659                 return file_exists($params['resource_name']);
01660             }
01661         } elseif (empty($this->_plugins['resource'][$params['resource_type']])) {
01662             $_params = array('type' => $params['resource_type']);
01663             require_once(SMARTY_CORE_DIR . 'core.load_resource_plugin.php');
01664             smarty_core_load_resource_plugin($_params, $this);
01665         }
